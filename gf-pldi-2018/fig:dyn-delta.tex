\fbox{Language $\langD$}\\
$\begin{array}{l@{~~}c@{~~}l}
  v & \Coloneqq & \ldots \mid \vlam{x}{e}
\\
  A & \Coloneqq & \ldots \mid \typeerror
\\
  \GammaD & \Coloneqq & \cdot \mid x,\GammaD
\\
  \ED & \Coloneqq & \holeD \mid \ED\,e \mid v\,\ED \mid \vpair{\ED}{e} \mid \vpair{v}{\ED} \mid
\\ & & \vunop\,\ED \mid \vbinop\,\ED\,e \mid \vbinop\,v\,\ED
\end{array}$

%% -----------------------------------------------------------------------------
\medskip
\fbox{$\GammaD \welldyn e$}
\begin{mathpar}
  \inferrule*{
    x \in \GammaD
  }{
    \GammaD \welldyn x
  }

  \inferrule*{
    x,\GammaD \welldyn e
  }{
    \GammaD \welldyn \vlam{x}{e}
  }

  \inferrule*{
  }{
    \GammaD \welldyn i
  }

  \inferrule*{
    \GammaD \welldyn e_0
    \\
    \GammaD \welldyn e_1
  }{
    \GammaD \welldyn \vpair{e_0}{e_1}
  }

  \inferrule*{
    \GammaD \welldyn e_0
    \\
    \GammaD \welldyn e_1
  }{
    \GammaD \welldyn e_0~e_1
  }

  \inferrule*{
    \GammaD \welldyn e
  }{
    \GammaD \welldyn \vunop~e
  }

  \inferrule*{
    \GammaD \welldyn e_0
    \\
    \GammaD \welldyn e_1
  }{
    \GammaD \welldyn \vbinop~e_0~e_1
  }
\end{mathpar}

\medskip
\fbox{$e \rrD A$}\\
$\begin{array}{l@{~~}c@{~~}l l}
  (\vlam{x}{e})~v & \rrD & \vsubst{e}{x}{v} &
\\
  v_0~v_1 & \rrD & \typeerror & \mbox{if $v_0 \neq \vlam{x}{e}$}
\\
  \eunop{v} & \rrD & A & \mbox{if $A = \delta(\vunop, v)$}
\\
  \eunop{v} & \rrD & \tagerror & \mbox{otherwise}
\\
  \ebinop{v_0}{v_1} & \rrD & A & \mbox{if $A = \delta(\vbinop, v_0, v_1)$}
\\
  \ebinop{v_0}{v_1} & \rrD & \tagerror & \mbox{otherwise}
\end{array}$
