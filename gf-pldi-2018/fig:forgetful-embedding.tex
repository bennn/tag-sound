\fbox{$\wellSD{v}{\tau}{v}$}
\begin{mathpar}
  \inferrule*{
    v \in \naturals
  }{
    \wellSD{v}{\tnat}{v}
  }

  \inferrule*{
    v \in \integers
  }{
    \wellSD{v}{\tint}{v}
  }

  \inferrule*{
    v = \vpair{v_0}{v_1}
  }{
    \wellSD{v}{(\tpair{\tau_0}{\tau_1})}{\vmonpair{(\tpair{\tau_0}{\tau_1})}{v}}
  }

  \inferrule*{
    v = \vmonpair{\tau}{v'}
  }{
    \wellSD{v}{(\tpair{\tau_0}{\tau_1})}{\vmonpair{(\tpair{\tau_0}{\tau_1})}{v'}}
  }

  \inferrule*{
    v = \vlam{\tann{x}{\tau}}{e}
    \\
    \vee
    \\
    v = \vlam{x}{e}
  }{
    \wellSD{v}{(\tarr{\tau_d}{\tau_c})}{\vmonfun{(\tarr{\tau_d}{\tau_c})}{v}}
  }

  \inferrule*{
    v = \vmonfun{\tau}{v'}
  }{
    \wellSD{v}{(\tarr{\tau_d}{\tau_c})}{\vmonfun{(\tarr{\tau_d}{\tau_c})}{v'}}
  }
\end{mathpar}
