\medskip
\fbox{$\Gamma \welldyn e$}
\begin{mathpar}
  \inferrule*{
    x \in \Gamma
  }{
    \Gamma \welldyn x
  }

  \inferrule*{
    x,\Gamma \welldyn e
  }{
    \Gamma \welldyn \vlam{x}{e}
  }

  \inferrule*{
  }{
    \Gamma \welldyn i
  }

  \inferrule*{
    \Gamma \welldyn e_0
    \\
    \Gamma \welldyn e_1
  }{
    \Gamma \welldyn \vpair{e_0}{e_1}
  }

  \inferrule*{
    \Gamma \welldyn e_0
    \\
    \Gamma \welldyn e_1
  }{
    \Gamma \welldyn e_0~e_1
  }

  \inferrule*{
    \Gamma \welldyn e
  }{
    \Gamma \welldyn \vunop~e
  }

  \inferrule*{
    \Gamma \welldyn e_0
    \\
    \Gamma \welldyn e_1
  }{
    \Gamma \welldyn \vbinop~e_0~e_1
  }

\end{mathpar}

\medskip

\fbox{$\delta(\mathsf{op}, v, \ldots) = A$}
$\begin{array}{l c l}
  \delta(\vfst, \vpair{v_0}{v_1}) & = & v_0
\\
  \delta(\vfst, v) & = & \typeerror
\\[0.8ex]
  \delta(\vsnd, \vpair{v_0}{v_1}) & = & v_1
\\
  \delta(\vsnd, v) & = & \typeerror
\\[0.8ex]
  \delta(\vsum, i_0, i_1) & = & i_0 + i_1
\\
  \delta(\vsum, v_0, v_1) & = & \typeerror
\\[0.8ex]
  \delta(\vquotient, i_0, i_1) & = & \tagof{i_0 / i_1}
\\
  \delta(\vquotient, i_0, 0) & = & \valueerror
\\
  \delta(\vquotient, v_0, v_1) & = & \typeerror
\end{array}$

\fbox{$e~~\rrD~~A$}
\begin{mathpar}
  \inferrule*{
  }{
    (\vlam{x}{e})~v~~\rrD~~\vsubst{e}{x}{v}
  }

  \inferrule*{
    v_0 \neq \vlam{x}{e}
  }{
    v_0~v_1~~\rrD~~\typeerror
  }

  \inferrule*{
  }{
    \vunop~v~~\rrD~~\delta(\vunop, v)
  }

  \inferrule*{
  }{
    \vbinop~v_0~v_1~~\rrD~~\delta(\vbinop, v_0, v_1)
  }
\end{mathpar}
