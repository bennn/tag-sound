\fbox{$\Gamma \wellKE e$}
\begin{mathpar}
  \inferrule*{
  }{
    \Gamma \wellKE i
  }

  \inferrule*{
    \Gamma \wellKE e_0
    \\
    \Gamma \wellKE e_1
  }{
    \Gamma \wellKE \vpair{e_0}{e_1}
  }

  \inferrule*{
    x,\Gamma \wellKE e
  }{
    \Gamma \wellKE \vlam{x}{e}
  }

  \inferrule*{
    \tann{x}{\tau},\Gamma \wellKE e : \kany
  }{
    \Gamma \wellKE \vlam{\tann{x}{\tau}}{e}
  }

  \inferrule*{
    x \in \Gamma
  }{
    \Gamma \wellKE x
  }

  \inferrule*{
    \tann{x}{\tau} \in \Gamma
  }{
    \Gamma \wellKE x
  }

  \inferrule*{
    \Gamma \wellKE e_0
    \\
    \Gamma \wellKE e_1
  }{
    \Gamma \wellKE e_0~e_1
  }

  \inferrule*{
    \Gamma \wellKE e
  }{
    \Gamma \wellKE \vunop~e
  }

  \inferrule*{
    \Gamma \wellKE e_0
    \\
    \Gamma \wellKE e_1
  }{
    \Gamma \wellKE \vbinop~e_0~e_1
  }

  \inferrule*{
    \Gamma \wellKE e : \tagof{\tau}
  }{
    \Gamma \wellKE \esta~\tau~e
  }

  \inferrule*{
    \Gamma \wellKE e : \kany
  }{
    \Gamma \wellKE \esta~\kany~e
  }

\end{mathpar}

\fbox{$\Gamma \wellKE e : K$}
\begin{mathpar}
  \inferrule*{
    i \in \naturals
  }{
    \Gamma \wellKE i : \knat
  }

  \inferrule*{
  }{
    \Gamma \wellKE i : \kint
  }

  \inferrule*{
    \Gamma \wellKE e_0 : \kany
    \\\\
    \Gamma \wellKE e_1 : \kany
  }{
    \Gamma \wellKE \vpair{e_0}{e_1} : \kpair
  }

  \inferrule*{
    x,\Gamma \wellKE e
  }{
    \Gamma \wellKE \vlam{x}{e} : \kfun
  }

  \inferrule*{
    \tann{x}{\tau},\Gamma \wellKE e : \kany
  }{
    \Gamma \wellKE \vlam{\tann{x}{\tau}}{e} : \kfun
  }

  \inferrule*{
    x \in \Gamma
  }{
    \Gamma \wellKE x : \kany
  }

  \inferrule*{
    \tann{x}{\tau} \in \Gamma
    \\\\
    \tagof{\tau} = K
  }{
    \Gamma \wellKE x : K
  }

  \inferrule*{
    \Gamma \wellKE e_0 : \kfun
    \\\\
    \Gamma \wellKE e_1 : \kany
  }{
    \Gamma \wellKE e_0~e_1 : \kany
  }

  \inferrule*{
    \Gamma \wellKE e : \kpair
  }{
    \Gamma \wellKE \efst{e} : \kany
  }

  \inferrule*{
    \Gamma \wellKE e : \kpair
  }{
    \Gamma \wellKE \esnd{e} : \kany
  }

  \inferrule*{
    \Gamma \wellKE e_0 : K_0
    \\
    \Gamma \wellKE e_1 : K_1
    %\\\\
    %\tau_0 = K_0
    %\\
    %\tau_1 = K_1
    \\\\
    \Delta(\vbinop, K_0, K_1) = K
  }{
    \Gamma \wellKE \vbinop~e_0~e_1 : K
  }

  \inferrule*{
    \Gamma \wellKE e
    \\
    \tagof{\tau} = K
  }{
    \Gamma \wellKE \edyn{\tau}{e} : K
  }

  \inferrule*{
    \Gamma \wellKE e
  }{
    \Gamma \wellKE \edyn{\kany}{e} : \kany
  }

  \inferrule*{
    \Gamma \wellKE e : \kany
  }{
    \Gamma \wellKE \echk{K}{e} : K
  }

  \inferrule*{
    \Gamma \wellKE e : K'
    \\
    K' \subk K
  }{
    \Gamma \wellKE e : K
  }
\end{mathpar}
