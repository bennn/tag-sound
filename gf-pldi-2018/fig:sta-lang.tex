\fbox{Language $\langS$}
$\begin{array}{l c l}
  e & \Coloneqq & x \mid v \mid e~e \mid \vpair{e}{e} \mid \vunop~e \mid \vbinop~e~e
\\
  v & \Coloneqq & \vlam{\tann{x}{\tau}}{e} \mid i \mid \vpair{v}{v}
\\
  \tau & \Coloneqq & \tnat \mid \tint \mid \tpair{\tau}{\tau} \mid \tau \tarrow \tau
\\
  i & \Coloneqq & 0 \mid 1 \mid -1 \mid \ldots
\\
  \vunop & \Coloneqq & \vfst \mid \vsnd
\\
  \vbinop & \Coloneqq & \vsum \mid \vquotient
\\
  \GammaS & \Coloneqq & \cdot \mid \tann{x}{\tau},\GammaS
\\
  A & \Coloneqq & e \mid \valueerror
\\
  E & \Coloneqq & \bullet \mid E~e \mid v~E \mid
                  \vpair{E}{e} \mid \vpair{v}{E} \mid \vunop~E \mid
\\ & &            \ebinop{E}{e} \mid \ebinop{v}{E}
\end{array}$

\medskip
\fbox{$\tau \subt \tau$}
\begin{mathpar}
  \inferrule*{
  }{
    \tnat \subt \tint
  }

  \inferrule*{
    \tau_d' \subt \tau_d
    \\\\
    \tau_c \subt \tau_c'
  }{
    \tarr{\tau_d}{\tau_c} \subt \tarr{\tau_d'}{\tau_c'}
  }

  \inferrule*{
    \tau_0 \subt \tau_0'
    \\\\
    \tau_1 \subt \tau_1'
  }{
    \tpair{\tau_0}{\tau_1} \subt \tpair{\tau_0'}{\tau_1'}
  }
\end{mathpar}

%% -----------------------------------------------------------------------------
\medskip
\fbox{$\Delta(\vunop, \tau) = \tau$}
\begin{mathpar}
  \inferrule*{
  }{
    \Delta(\vfst, \tpair{\tau_0}{\tau_1}) = \tau_0
  }

  \inferrule*{
  }{
    \Delta(\vsnd, \tpair{\tau_0}{\tau_1}) = \tau_1
  }
\end{mathpar}

\fbox{$\Delta(\vunop, \tau, \tau) = \tau$}
\begin{mathpar}
  \inferrule*{
  }{
    \Delta(\vbinop, \tnat, \tnat) = \tnat
  }

  \inferrule*{
    \tau_0 \subt \tint
    \\
    \tau_1 \subt \tint
  }{
    \Delta(\vbinop, \tau_0, \tau_1) = \tint
  }
\end{mathpar}

%% -----------------------------------------------------------------------------
\medskip
\fbox{$\GammaS \wellsta e : \tau$}
\begin{mathpar}
  \inferrule*{
    \tann{x}{\tau} \in \GammaS
  }{
    \GammaS \wellsta x : \tau
  }

  \inferrule*{
    \tann{x}{\tau_d},\GammaS \wellsta e : \tau_c
  }{
    \GammaS \wellsta \vlam{\tann{x}{\tau_d}}{e} : \tau_d \tarrow \tau_c
  }

  \inferrule*{
    i \in \naturals
  }{
    \GammaS \wellsta i : \tnat
  }

  \inferrule*{
    i \in \integers
  }{
    \GammaS \wellsta i : \tint
  }

  \inferrule*{
    \GammaS \wellsta e_0 : \tau_0
    \\\\
    \GammaS \wellsta e_1 : \tau_1
  }{
    \GammaS \wellsta \vpair{e_0}{e_1} : \tpair{\tau_0}{\tau_1}
  }

  \inferrule*{
    \GammaS \wellsta e_0 : \tau_d \tarrow \tau_c
    \\\\
    \GammaS \wellsta e_1 : \tau_d
  }{
    \GammaS \wellsta e_0~e_1 : \tau_c
  }

  \inferrule*{
    \GammaS \wellsta e_0 : \tau_0
    \\\\
    \Delta(\vunop, \tau_0) = \tau
  }{
    \GammaS \wellsta \vunop~e_0 : \tau
  }

  \inferrule*{
    \GammaS \wellsta e_0 : \tau_0
    \\\\
    \GammaS \wellsta e_1 : \tau_1
    \\\\
    \Delta(\vbinop, \tau_0, \tau_1) = \tau
  }{
    \GammaS \wellsta \vbinop~e_0~e_1 : \tau
  }

  \inferrule*{
    \GammaS \wellsta e : \tau'
    \\\\
    \tau' \subt \tau
  }{
    \GammaS \wellsta e : \tau
  }
\end{mathpar}

\medskip
\fbox{$\deltaS(\vunop, v) = A$}\\
$\begin{array}{l c l}
  \deltaS(\vfst, \vpair{v_0}{v_1}) & = & v_0
\\
  \deltaS(\vsnd, \vpair{v_0}{v_1}) & = & v_1
\end{array}$

\medskip
\fbox{$\deltaS(\vbinop, v, v) = A$}
$\begin{array}{l c l}
  \deltaS(\vsum, i_0, i_1) & = & i_0 + i_1
\\
  \deltaS(\vquotient, i_0, 0) & = & \valueerror
\\
  \deltaS(\vquotient, i_0, i_1) & = & \tagof{i_0 / i_1}
\end{array}$

\medskip
\fbox{$e \rrS A$}
\begin{mathpar}
  \inferrule*{
  }{
    (\vlam{x:\tau}{e})~v \rrS \vsubst{e}{x}{v}
  }

  \inferrule*{
    A = \deltaS(\vunop, v)
  }{
    \vunop~v \rrS A
  }

  \inferrule*{
    A = \deltaS(\vbinop, v_0, v_1)
  }{
    \vbinop~v_0~v_1 \rrS A
  }
\end{mathpar}

\medskip
$\begin{array}{l c l}
  \ccS & \triangleq & \mbox{context closure of $\rrS$ in $E$}
\\
  \rrSstar & \triangleq & \mbox{reflexive, transitive closure of $\ccS$}
\end{array}$
