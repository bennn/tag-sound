\fbox{Language $\langE$} extends $\langM$:
$\begin{array}{l c l}
  E & \Coloneqq & \bullet \mid E~e \mid v~E \mid
                  \vpair{E}{e} \mid \vpair{v}{E} \mid \vunop~E \mid
\\ & &            \ebinop{E}{e} \mid \ebinop{v}{E} \mid \edyn{\tau}{E} \mid \esta{\tau}{E}
\end{array}$

%% -----------------------------------------------------------------------------
\medskip
\fbox{$\Gamma \wellEE e$} extends $\Gamma \welldyn e$:\\
\begin{mathpar}
  \inferrule*{
    x,\Gamma \wellEE e
  }{
    \Gamma \wellEE \vlam{\tann{x}{\tau}}{e}
  }

  \inferrule*{
    \Gamma \wellEE e
  }{
    \Gamma \wellEE \edyn{\tau}{e}
  }

  \inferrule*{
    \Gamma \wellEE e
  }{
    \Gamma \wellEE \esta{\tau}{e}
  }
\end{mathpar}

%% -----------------------------------------------------------------------------
\medskip
\fbox{$e \rrEE A$} extends $e \rrD A$:\\
\begin{mathpar}
  \inferrule*{
  }{
    \edyn{\tau}{v} \rrEE v
  }

  \inferrule*{
  }{
    \esta{\tau}{v} \rrEE v
  }

  \inferrule*{
  }{
    (\vlam{\tann{x}{\tau}}{e})~v \rrEE \vsubst{e}{x}{v}
  }
\end{mathpar}

%% -----------------------------------------------------------------------------

\end{array}$
