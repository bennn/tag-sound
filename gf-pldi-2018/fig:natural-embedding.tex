\begin{flushleft}

\fbox{Language $\langN$} extends $\langM$:\\
$\begin{array}{l c l}
  v & = & \ldots \mid \vmonfun{(\tau\tarrow\tau)}{v}
\\
  E & \Coloneqq & \bullet \mid E~e \mid v~E \mid
                  \vpair{E}{e} \mid \vpair{v}{E} \mid \vunop~E \mid
\\ & &            \ebinop{E}{e} \mid \ebinop{v}{E}
\end{array}$

%% -----------------------------------------------------------------------------
\medskip
\begin{minipage}[t]{0.5\textwidth}
\fbox{$\Gamma \wellNE e$} extends $\Gamma \wellM e$:
\begin{mathpar}
  \inferrule*{
    \Gamma \wellNE v : \tarr{\tau_d}{\tau_c}
  }{
    \Gamma \wellNE \vmonfun{(\tarr{\tau_d}{\tau_c})}{v}
  }
\end{mathpar}
\end{minipage}%
\begin{minipage}[t]{0.5\textwidth}
\fbox{$\Gamma \wellNE e : \tau$} extends $\Gamma \wellM e : \tau$:
\begin{mathpar}
  \inferrule*{
    \Gamma \wellNE v
  }{
    \Gamma \wellNE \vmonfun{(\tarr{\tau_d}{\tau_c})}{v} : (\tarr{\tau_d}{\tau_c})
  }
\end{mathpar}
\end{minipage}

%% -----------------------------------------------------------------------------
\medskip
\begin{minipage}[t]{0.5\textwidth}
\fbox{$e \rrND A$} extends $e \rrD A$:
\begin{mathpar}
  \inferrule*{
    e_{arg} = (\edyn{\tau_d}{v_{arg}})
  }{
    (\vmonfun{(\tau_d\tarrow\tau_c)}{v_f})~v_{arg} \rrND \esta{\tau_c}{(v_f~e_{arg})}
  }
\end{mathpar}
\end{minipage}%
\begin{minipage}[t]{0.5\textwidth}
\fbox{$e \rrNS A$} extends $e \rrS A$:
\begin{mathpar}
  \inferrule*{
    e_{arg} = (\esta{\tau_d}{v_{arg}})
  }{
    (\vmonfun{(\tau_d\tarrow\tau_c)}{v_f})~v_{arg} \rrNS \edyn{\tau_c}{(v_f~e_{arg})}
  }
\end{mathpar}
\end{minipage}%

%% -----------------------------------------------------------------------------
\medskip
\fbox{$\wellSD{v}{\tau}{v}$}
\begin{mathpar}
  \inferrule*{
  }{
    \wellSD{v}{\tnat}{v}
  }

  \inferrule*{
  }{
    \wellSD{v}{\tint}{v}
  }

  \inferrule*{
    \wellSD{v_0}{\tau_0}{v_0'}
    \\
    \wellSD{v_1}{\tau_1}{v_1'}
  }{
    \wellSD{\vpair{v_0}{v_1}}{(\tpair{\tau_0}{\tau_1})}{\vpair{v_0'}{v_1'}}
  }

  \inferrule*{
  }{
    \wellSD{v}{(\tarr{\tau_d}{\tau_c})}{\vmonfun{(\tarr{\tau_d}{\tau_c})}{v}}
  }
\end{mathpar}

\medskip
$\fromsta(\tau, v) = \begin{array}{l l}
  v' & \mbox{where $\wellSD{v}{\tau}{v'}$}
\end{array}$

%% -----------------------------------------------------------------------------
\medskip
\fbox{$\wellDS{v}{\tau}{v}$}
\begin{mathpar}
  \inferrule*{
    v \in \naturals
  }{
    \wellDS{v}{\tnat}{v}
  }

  \inferrule*{
    v \in \integers
  }{
    \wellDS{v}{\tint}{v}
  }

  \inferrule*{
    \wellDS{v_0}{\tau_0}{v_0'}
    \\\\
    \wellDS{v_1}{\tau_1}{v_1'}
  }{
    \wellDS{\vpair{v_0}{v_1}}{(\tpair{\tau_0}{\tau_1})}{\vpair{v_0'}{v_1'}}
  }

  \inferrule*{
    v = \vlam{x}{e}
    \\
    \vee
    \\
    v = \vmonfun{\tau}{v'}
  }{
    \wellDS{v}{(\tarr{\tau_d}{\tau_c})}{\vmonfun{(\tarr{\tau_d}{\tau_c})}{v}}
  }
\end{mathpar}

\medskip
$\fromdyn(\tau, v) = \left\{\begin{array}{l l}
  v' & \mbox{if $\wellDS{v}{\tau}{v'}$}
\\
  \valueerror & \mbox{otherwise}
\end{array}\right.$

\medskip
\begin{minipage}[t]{0.5\textwidth}
\fbox{$e \ccND A$}
\begin{mathpar}
  \inferrule*{
    \fromsta(\tau, v) = v'
  }{
    E[\esta{\tau}{v}] \ccND E[v']
  }

  \inferrule*{
    e \ccNS e'
  }{
    E[\esta{\tau}{e}] \ccND E[\esta{\tau}{e'}]
  }

  \inferrule*{
    e \ccNS A
    \\\\
    A = \typeerror
    \\
    \vee
    \\
    A = \valueerror
  }{
    E[\esta{\tau}{e}] \ccND A
  }

  \inferrule*{
    e \rrND e'
  }{
    E[e] \ccND E[e']
  }

  \inferrule*{
    e \rrND A
    \\\\
    A = \typeerror
    \\
    \vee
    \\
    A = \valueerror
  }{
    E[e] \ccND A
  }

\end{mathpar}
\end{minipage}%
\begin{minipage}[t]{0.5\textwidth}
\fbox{$e \ccNS A$}
\begin{mathpar}
  \inferrule*{
    \fromdyn(\tau, v) = v'
  }{
    E[\edyn{\tau}{v}] \ccNS E[v']
  }

  \inferrule*{
    \fromdyn(\tau, v) = \valueerror
  }{
    E[\edyn{\tau}{v}] \ccNS \valueerror
  }

  \inferrule*{
    e \ccND e'
  }{
    E[\edyn{\tau}{e}] \ccNS E[e']
  }

  \inferrule*{
    e \ccND A
    \\\\
    A = \typeerror
    \\
    \vee
    \\
    A = \valueerror
  }{
    E[\edyn{\tau}{e}] \ccNS A
  }

  \inferrule*{
    e \rrNS e'
  }{
    E[e] \ccNS E[e']
  }

  \inferrule*{
    e \rrNS A
    \\\\
    A = \typeerror
    \\
    \vee
    \\
    A = \valueerror
  }{
    E[e] \ccNS A
  }
\end{mathpar}
\end{minipage}
