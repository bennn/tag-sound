\begin{mathpar}
  \inferrule*{
    v \neq \vlam{(x:\tau)}{e}
  }{
    \esta{\tau}{v} \dynstep v
  }

  \inferrule*{
  }{
    \esta{\tau}{\vlam{(x:\tau)}{e}} \dynstep \valueerror
  }

  \inferrule*{
    i \in \integers
  }{
    \edyn{\tint}{i} \stastep i
  }

  \inferrule*{
    v \not\in \integers
  }{
    \edyn{\tint}{v} \stastep \valueerror
  }

  \inferrule*{
    i \in \naturals
  }{
    \edyn{\tnat}{i} \stastep i
  }

  \inferrule*{
    v \not\in \naturals
  }{
    \edyn{\tnat}{v} \stastep \valueerror
  }

  \inferrule*{
  }{
    \edyn{\tpair{\tau}{\tau'}}{\vpair{v}{v'}} \stastep \vpair{\edyn{\tau}{v}}{\edyn{\tau'}{v'}}
  }

  \inferrule*{
    v \neq \vpair{v'}{v''}
  }{
    \edyn{\tpair{\tau}{\tau'}}{v} \stastep \valueerror
  }

\end{mathpar}
