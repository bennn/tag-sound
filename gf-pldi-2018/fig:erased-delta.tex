{\flushleft

$\fromdyn(\tau, v) = v$

\smallskip

$\fromsta(\tau, v) = v$

\medskip
\fbox{$e \rrEE A$} extends $e \rrD A$:
\begin{mathpar}
  \inferrule*{
  }{
    (\vlam{\tann{x}{\tau}}{e})~v \rrEE \vsubst{e}{x}{v}
  }

  \inferrule*{
    \fromdyn(\tau, v) = v
  }{
    \edyn{\tau}{v} \rrEE v
  }

  \inferrule*{
    \fromsta(\tau, v) = v
  }{
    \esta{\tau}{v} \rrEE v
  }
\end{mathpar}

\medskip
\fbox{$\Gamma \wellEE e$} extends $\Gamma \wellD e$:
\begin{mathpar}
  \inferrule*{
    x,\Gamma \wellEE e
  }{
    \Gamma \wellEE \vlam{\tann{x}{\tau}}{e}
  }

  \inferrule*{
    \Gamma \wellEE e
  }{
    \Gamma \wellEE \edyn{\tau}{e}
  }

  \inferrule*{
    \Gamma \wellEE e
  }{
    \Gamma \wellEE \esta{\tau}{e}
  }
\end{mathpar}

}
