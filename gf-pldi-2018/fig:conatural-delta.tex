\fbox{Language $\langL$} inherits $\langN$:\\
$\begin{array}{l@{~~}c@{~~}l}
  v & \Coloneqq & \ldots \mid \vmonpair{(\tpair{\tau}{\tau})}{v}
\end{array}$

\begin{mathpar}
  \inferrule*{
    \Gamma \wellLE v : \tpair{\tau_0}{\tau_1}
  }{
    \Gamma \wellLE \vmonpair{(\tpair{\tau_0}{\tau_1})}{v}
  }
\end{mathpar}
\end{minipage}%
\begin{minipage}[t]{\columnwidth}
\fbox{$\Gamma \wellLE e : \tau$} inherits $\Gamma \wellNE e : \tau$:
\begin{mathpar}
  \inferrule*{
    \Gamma \wellLE v
  }{
    \Gamma \wellLE \vmonfun{(\tpair{\tau_0}{\tau_1})}{v} : (\tarr{\tau_d}{\tau_c})
  }
\end{mathpar}
\end{minipage}
