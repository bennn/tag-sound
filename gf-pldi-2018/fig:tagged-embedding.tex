\medskip
\fbox{$\vchk(K, v) = A$}
\begin{mathpar}
  \inferrule*{
    i \in \naturals
  }{
    \vchk(\knat, i) = i
  }

  \inferrule*{
    i \in \integers
  }{
    \vchk(\kint, i) = i
  }

  \inferrule*{
    v = \vpair{v_0}{v_1}
  }{
    \vchk(\kpair, v) = v
  }

  \inferrule*{
    v = \vlam{x}{e}
    \\\\
    \vee
    \\\\
    v = \vlam{\tann{x}{\tau}}{e}
  }{
    \vchk(\kfun, v) = v
  }

  \inferrule*{
  }{
    \vchk(\kany, v) = v
  }
\end{mathpar}

\medskip
$\fromsta(\tau, v) = \left\{\begin{array}{l l}
  v' & \mbox{if $\vchk(\tagof{\tau}, v) = v'$}
\end{array}\right.$

\medskip
$\fromdyn(\tau, v) = \left\{\begin{array}{l l}
  v' & \mbox{if $\vchk(\tagof{\tau}, v) = v'$}
\\
  \valueerror & \mbox{otherwise}
\end{array}\right.$

\fbox{$\Gamma \wellKE e$}
\begin{mathpar}
  \inferrule*{
  }{
    \Gamma \wellKE i
  }

  \inferrule*{
    \Gamma \wellKE e_0
    \\
    \Gamma \wellKE e_1
  }{
    \Gamma \wellKE \vpair{e_0}{e_1}
  }

  \inferrule*{
    x,\Gamma \wellKE e
  }{
    \Gamma \wellKE \vlam{x}{e}
  }

  \inferrule*{
    \tann{x}{\tau},\Gamma \wellKE e : \kany
  }{
    \Gamma \wellKE \vlam{\tann{x}{\tau}}{e}
  }

  \inferrule*{
    x \in \Gamma
  }{
    \Gamma \wellKE x
  }

  \inferrule*{
    \tann{x}{\tau} \in \Gamma
  }{
    \Gamma \wellKE x
  }

  \inferrule*{
    \Gamma \wellKE e_0
    \\
    \Gamma \wellKE e_1
  }{
    \Gamma \wellKE e_0~e_1
  }

  \inferrule*{
    \Gamma \wellKE e
  }{
    \Gamma \wellKE \vunop~e
  }

  \inferrule*{
    \Gamma \wellKE e_0
    \\
    \Gamma \wellKE e_1
  }{
    \Gamma \wellKE \vbinop~e_0~e_1
  }

  \inferrule*{
    \Gamma \wellKE e : \tagof{\tau}
  }{
    \Gamma \wellKE \esta~\tau~e
  }

\end{mathpar}

\medskip
\fbox{$\Gamma \wellKE e : K$}
\begin{mathpar}
  \inferrule*{
    i \in \naturals
  }{
    \Gamma \wellKE i : \knat
  }

  \inferrule*{
  }{
    \Gamma \wellKE i : \kint
  }

  \inferrule*{
    \Gamma \wellKE e_0 : \kany
    \\\\
    \Gamma \wellKE e_1 : \kany
  }{
    \Gamma \wellKE \vpair{e_0}{e_1} : \kpair
  }

  \inferrule*{
    x,\Gamma \wellKE e
  }{
    \Gamma \wellKE \vlam{x}{e} : \kfun
  }

  \inferrule*{
    \tann{x}{\tau},\Gamma \wellKE e : \kany
  }{
    \Gamma \wellKE \vlam{\tann{x}{\tau}}{e} : \kfun
  }

  \inferrule*{
    x \in \Gamma
  }{
    \Gamma \wellKE x : \kany
  }

  \inferrule*{
    \tann{x}{\tau} \in \Gamma
    \\\\
    \tagof{\tau} = K
  }{
    \Gamma \wellKE x : K
  }

  \inferrule*{
    \Gamma \wellKE e_0 : \kfun
    \\\\
    \Gamma \wellKE e_1 : \kany
  }{
    \Gamma \wellKE e_0~e_1 : \kany
  }

  \inferrule*{
    \Gamma \wellKE e : \kpair
  }{
    \Gamma \wellKE \efst{e} : \kany
  }

  \inferrule*{
    \Gamma \wellKE e : \kpair
  }{
    \Gamma \wellKE \esnd{e} : \kany
  }

  \inferrule*{
    \Gamma \wellKE e_0 : K_0
    \\\\
    \Gamma \wellKE e_1 : K_1
    \\\\
    \Delta(\vbinop, K_0, K_1) = K
  }{
    \Gamma \wellKE \vbinop~e_0~e_1 : K
  }

  \inferrule*{
    \Gamma \wellKE e
    \\\\
    \tagof{\tau} = K
  }{
    \Gamma \wellKE \edyn{\tau}{e} : K
  }

  \inferrule*{
    \Gamma \wellKE e : K'
  }{
    \Gamma \wellKE \echk{K}{e} : K
  }

  \inferrule*{
    \Gamma \wellKE e : K'
    \\\\
    K' \subk K
  }{
    \Gamma \wellKE e : K
  }
\end{mathpar}
