\medskip
\fbox{$\Gamma \wellKE e$}
\begin{mathpar}
  \inferrule*{
  }{
    \Gamma \wellKE i
  }

  \inferrule*{
    \Gamma \wellKE e_0
    \\
    \Gamma \wellKE e_1
  }{
    \Gamma \wellKE \vpair{e_0}{e_1}
  }

  \inferrule*{
    x,\Gamma \wellKE e
  }{
    \Gamma \wellKE \vlam{x}{e}
  }

  \inferrule*{
    \tann{x}{\tau},\Gamma \wellKE e : \kany
  }{
    \Gamma \wellKE \vlam{\tann{x}{\tau}}{e}
  }

  \inferrule*{
    x \in \Gamma
  }{
    \Gamma \wellKE x
  }

  \inferrule*{
    \tann{x}{\tau} \in \Gamma
  }{
    \Gamma \wellKE x
  }

  \inferrule*{
    \Gamma \wellKE e_0
    \\
    \Gamma \wellKE e_1
  }{
    \Gamma \wellKE e_0~e_1
  }

  \inferrule*{
    \Gamma \wellKE e
  }{
    \Gamma \wellKE \vunop~e
  }

  \inferrule*{
    \Gamma \wellKE e_0
    \\
    \Gamma \wellKE e_1
  }{
    \Gamma \wellKE \vbinop~e_0~e_1
  }

  \inferrule*{
    \Gamma \wellKE e : \tagof{\tau}
  }{
    \Gamma \wellKE \esta~\tau~e
  }

  \inferrule*{
    \Gamma \wellKE e : \kany
  }{
    \Gamma \wellKE \esta~\kany~e
  }

\end{mathpar}

\medskip
\fbox{$\Gamma \wellKE e : K$}
\begin{mathpar}
  \inferrule*{
    i \in \naturals
  }{
    \Gamma \wellKE i : \knat
  }

  \inferrule*{
  }{
    \Gamma \wellKE i : \kint
  }

  \inferrule*{
    \Gamma \wellKE e_0 : \kany
    \\\\
    \Gamma \wellKE e_1 : \kany
  }{
    \Gamma \wellKE \vpair{e_0}{e_1} : \kpair
  }

  \inferrule*{
    x,\Gamma \wellKE e
  }{
    \Gamma \wellKE \vlam{x}{e} : \kfun
  }

  \inferrule*{
    \tann{x}{\tau},\Gamma \wellKE e : \kany
  }{
    \Gamma \wellKE \vlam{\tann{x}{\tau}}{e} : \kfun
  }

  \inferrule*{
    x \in \Gamma
  }{
    \Gamma \wellKE x : \kany
  }

  \inferrule*{
    \tann{x}{\tau} \in \Gamma
    \\\\
    \tagof{\tau} = K
  }{
    \Gamma \wellKE x : K
  }

  \inferrule*{
    \Gamma \wellKE e_0 : \kfun
    \\\\
    \Gamma \wellKE e_1 : \kany
  }{
    \Gamma \wellKE e_0~e_1 : \kany
  }

  \inferrule*{
    \Gamma \wellKE e : \kpair
  }{
    \Gamma \wellKE \efst{e} : \kany
  }

  \inferrule*{
    \Gamma \wellKE e : \kpair
  }{
    \Gamma \wellKE \esnd{e} : \kany
  }

  \inferrule*{
    \Gamma \wellKE e_0 : K_0
    \\
    \Gamma \wellKE e_1 : K_1
    %\\\\
    %\tau_0 = K_0
    %\\
    %\tau_1 = K_1
    \\\\
    \Delta(\vbinop, \tau_0, \tau_1) = K
  }{
    \Gamma \wellKE \vbinop~e_0~e_1 : K
  }

  \inferrule*{
    \Gamma \wellKE e
    \\
    \tagof{\tau} = K
  }{
    \Gamma \wellKE \edyn{\tau}{e} : K
  }

  \inferrule*{
    \Gamma \wellKE e
  }{
    \Gamma \wellKE \edyn{\kany}{e} : \kany
  }

  \inferrule*{
    \Gamma \wellKE e : K'
  }{
    \Gamma \wellKE \echk{K}{e} : K
  }

  \inferrule*{
    \Gamma \wellKE e : K'
    \\
    K' \subk K
  }{
    \Gamma \wellKE e : K
  }
\end{mathpar}

\fbox{$e \rrKD A$} extends $e \rrD A$:
\begin{mathpar}
  \inferrule*{
    v = \deltaC(\tagof{\tau}, v)
  }{
    (\vlam{\tann{x}{\tau}}{e})~v \rrKD (\esta{\kany}{\vsubst{e}{x}{v}})
  }

  \inferrule*{
    \valueerror = \deltaC(\tagof{\tau}, v)
  }{
    (\vlam{\tann{x}{\tau}}{e})~v \rrKD \valueerror
  }

  \inferrule*{
  }{
    (\esta{\kany}{v}) \rrKD v
  }
\end{mathpar}

\medskip
\fbox{$e \rrKS A$} extends $e \rrS A$:
\begin{mathpar}
  \inferrule*{
  }{
    (\vlam{x}{e})~v \rrKS (\edyn{\kany}{\vsubst{e}{x}{v}})
  }

  \inferrule*{
  }{
    (\edyn{\kany}{v}) \rrKS v
  }

  \inferrule*{
    v = \deltaC{K}{v}
  }{
    \echk{K}{v} \rrKS v
  }

  \inferrule*{
    \valueerror = \deltaC{K}{v}
  }{
    (\echk{K}{v}) \rrKS \valueerror
  }
\end{mathpar}

\medskip
$\deltaC(K, v) = \left\{\begin{array}{l l}
  v & \mbox{if $\wellKE v : K$}
\\
  \valueerror & \mbox{otherwise}
\end{array}\right.$

\medskip
$\fromdyn(\tau, v) = \deltaC(\tagof{\tau}, v)$

\medskip
$\fromsta(\tau, v) = \deltaC(\tagof{\tau}, v)$
