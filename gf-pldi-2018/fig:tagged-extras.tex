\fbox{$\Gamma \wellKE e$}
\begin{mathpar}
  \inferrule*{
  }{
    \Gamma \wellKE i
  }

  \inferrule*{
    \Gamma \wellKE e_0
    \\\\
    \Gamma \wellKE e_1
  }{
    \Gamma \wellKE \vpair{e_0}{e_1}
  }

  \inferrule*{
    x,\Gamma \wellKE e
  }{
    \Gamma \wellKE \vlam{x}{e}
  }

  \inferrule*{
    \tann{x}{\tau},\Gamma \wellKE e : \kany
  }{
    \Gamma \wellKE \vlam{\tann{x}{\tau}}{e}
  }

  \inferrule*{
    x \in \Gamma
  }{
    \Gamma \wellKE x
  }

  \inferrule*{
    \tann{x}{\tau} \in \Gamma
  }{
    \Gamma \wellKE x
  }

  \inferrule*{
    \Gamma \wellKE e_0
    \\\\
    \Gamma \wellKE e_1
  }{
    \Gamma \wellKE e_0~e_1
  }

  \inferrule*{
    \Gamma \wellKE e
  }{
    \Gamma \wellKE \vunop~e
  }

  \inferrule*{
    \Gamma \wellKE e_0
    \\\\
    \Gamma \wellKE e_1
  }{
    \Gamma \wellKE \vbinop~e_0~e_1
  }

  \inferrule*{
    \Gamma \wellKE e : \tagof{\tau}
  }{
    \Gamma \wellKE \esta~\tau~e
  }

\end{mathpar}

\medskip
\fbox{$\Gamma \wellKE e : K$}
\begin{mathpar}
  \inferrule*{
    i \in \naturals
  }{
    \Gamma \wellKE i : \knat
  }

  \inferrule*{
  }{
    \Gamma \wellKE i : \kint
  }

  \inferrule*{
    \Gamma \wellKE e_0 : \kany
    \\\\
    \Gamma \wellKE e_1 : \kany
  }{
    \Gamma \wellKE \vpair{e_0}{e_1} : \kpair
  }

  \inferrule*{
    x,\Gamma \wellKE e
  }{
    \Gamma \wellKE \vlam{x}{e} : \kfun
  }

  \inferrule*{
    \tann{x}{\tau},\Gamma \wellKE e : \kany
  }{
    \Gamma \wellKE \vlam{\tann{x}{\tau}}{e} : \kfun
  }

  \inferrule*{
    x \in \Gamma
  }{
    \Gamma \wellKE x : \kany
  }

  \inferrule*{
    \tann{x}{\tau} \in \Gamma
    \\\\
    \tagof{\tau} = K
  }{
    \Gamma \wellKE x : K
  }

  \inferrule*{
    \Gamma \wellKE e_0 : \kfun
    \\\\
    \Gamma \wellKE e_1 : \kany
  }{
    \Gamma \wellKE e_0~e_1 : \kany
  }

  \inferrule*{
    \Gamma \wellKE e : \kpair
  }{
    \Gamma \wellKE \efst{e} : \kany
  }

  \inferrule*{
    \Gamma \wellKE e : \kpair
  }{
    \Gamma \wellKE \esnd{e} : \kany
  }

  \inferrule*{
    \Gamma \wellKE e_0 : K_0
    \\\\
    \Gamma \wellKE e_1 : K_1
    \\\\
    \Delta(\vbinop, K_0, K_1) = K
  }{
    \Gamma \wellKE \vbinop~e_0~e_1 : K
  }

  \inferrule*{
    \Gamma \wellKE e
    \\\\
    \tagof{\tau} = K
  }{
    \Gamma \wellKE \edyn{\tau}{e} : K
  }

  \inferrule*{
    \Gamma \wellKE e : K'
  }{
    \Gamma \wellKE \echk{K}{e} : K
  }

  \inferrule*{
    \Gamma \wellKE e : K'
    \\\\
    K' \subk K
  }{
    \Gamma \wellKE e : K
  }
\end{mathpar}

\fbox{$K \subk K$}
\begin{mathpar}
  \inferrule*{
  }{
    K \subk \kany
  }

  \inferrule*{
  }{
    \knat \subk \kint
  }

  \inferrule*{
  }{
    K \subk K
  }
\end{mathpar}

\fbox{$\Gamma \welldyn e \carrow e$}
\begin{mathpar}
  \inferrule*{
  }{
    \Gamma \welldyn i \carrow i
  }

  \inferrule*{
    \Gamma \welldyn e_0 \carrow e_0'
    \\\\
    \Gamma \welldyn e_1 \carrow e_1'
  }{
    \Gamma \welldyn \vpair{e_0}{e_1} \carrow \vpair{e_0'}{e_1'}
  }

  \inferrule*{
    x,\Gamma \welldyn e \carrow e'
  }{
    \Gamma \welldyn \vlam{x}{e} \carrow \vlam{x}{e'}
  }

  \inferrule*{
    \tann{x}{\tau_d},\Gamma \wellsta e : \tau_c \carrow e'
  }{
    \Gamma \welldyn \vlam{\tann{x}{\tau_d}}{e} \carrow \vlam{\tann{x}{\tau_d}}{e'}
  }

  \inferrule*{
  }{
    \Gamma \welldyn x \carrow x
  }

  \inferrule*{
    \Gamma \welldyn e_0 \carrow e_0'
    \\\\
    \Gamma \welldyn e_1 \carrow e_1'
  }{
    \Gamma \welldyn e_0~e_1 \carrow e_0'~e_1'
  }

  \inferrule*{
    \Gamma \wellsta e : \tau \carrow e'
  }{
    \Gamma \welldyn \esta{\tau}{e} \carrow \esta{\tau}{e'}
  }

\end{mathpar}
