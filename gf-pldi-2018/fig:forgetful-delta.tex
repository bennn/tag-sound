\fbox{Language $\langF$} extends $\langM$:\\
$\begin{array}{l@{~~}c@{~~}l}
  v & \Coloneqq & \ldots \mid \vmonfun{(\tarr{\tau}{\tau})}{(\vlam{x}{e})} \mid
      \vmonfun{(\tarr{\tau}{\tau})}{(\vlam{\tann{x}{\tau}}{e})} \mid
      \vmonpair{(\tpair{\tau}{\tau})}{\vpair{v}{v}}
\end{array}$

\bigskip
\begin{minipage}[t]{\columnwidth}
\fbox{$\Gamma \wellFE e$} inherits $\Gamma \wellM e$:
\begin{mathpar}
  \inferrule*{
  }{
    \Gamma \wellFE \vmonfun{(\tarr{\tau_d}{\tau_c})}{v}
  }

  \inferrule*{
  }{
    \Gamma \wellFE \vmonpair{(\tpair{\tau_0}{\tau_1})}{v}
  }
\end{mathpar}
\end{minipage}%
\begin{minipage}[t]{\columnwidth}
\fbox{$\Gamma \wellFE e : \tau$} inherits $\Gamma \wellM e : \tau$:
\begin{mathpar}
  \inferrule*{
  }{
    \Gamma \wellFE \vmonfun{(\tarr{\tau_d}{\tau_c})}{v} : \tarr{\tau_d}{\tau_c}
  }

  \inferrule*{
  }{
    \Gamma \wellFE \vmonpair{(\tpair{\tau_0}{\tau_1})}{v} : \tpair{\tau_0}{\tau_1}
  }
\end{mathpar}
\end{minipage}
