\begin{minipage}[t]{0.5\textwidth}
\fbox{Language $\langF$} extends $\langM$:\\
$\begin{array}{l@{~~}c@{~~}l}
  v & \Coloneqq & \ldots \mid \vmonfun{(\tarr{\tau}{\tau})}{\vlam{x}{e}} \mid
\\ & & \vmonfun{(\tarr{\tau}{\tau})}{\vlam{\tann{x}{\tau}}{e}} \mid
\\ & & \vmonpair{(\tpair{\tau}{\tau})}{\vpair{v}{v}}
\end{array}$

\medskip
\fbox{$\fromsta(\tau, v) = A$}\\
$\begin{array}{l}
  \fromsta(\tau, v) = \fromany(\tau, v)
\end{array}$

\medskip
\fbox{$\fromdyn(\tau, v) = A$}\\
$\begin{array}{l}
  \fromdyn(\tau, v) = \fromany(\tau, v)
\end{array}$
\end{minipage}%
\begin{minipage}[t]{0.5\textwidth}
\fbox{$\fromany(\tau, v) = A$}\\
$\begin{array}{l}
  \fromany(\tint, i) = i
\\
  \fromany(\tnat, i) = i
\\ \mbox{\quad if $i \in \naturals$}
\\
  \fromany(\tpair{\tau_0}{\tau_1}, v) = \vmonpair{(\tpair{\tau_0}{\tau_1})}{v}
\\ \mbox{\quad if $v = \vpair{v_0}{v_1}$}
\\
  \fromany(\tpair{\tau_0}{\tau_1}, v) = \vmonpair{(\tpair{\tau_0}{\tau_1})}{v'}
\\ \mbox{\quad if $v = \vmonpair{(\tpair{\tau_0'}{\tau_1'})}{v'}$}
\\
  \fromany(\tarr{\tau_d}{\tau_c}, v) = \vmonfun{(\tarr{\tau_d}{\tau_c})}{v}
\\ \mbox{\quad if $v = \vlam{x}{e}$ or $v = \vlam{\tann{x}{\tau}}{e}$}
\\
  \fromany(\tarr{\tau_d}{\tau_c}, v) = \vmonfun{(\tarr{\tau_d}{\tau_c})}{v'}
\\ \mbox{\quad if $v = \vmonfun{(\tarr{\tau_d'}{\tau_c'})}{v'}$}
\\
  \fromany(\tau, v) = \valueerror
\\ \mbox{\quad otherwise}
\end{array}$
\end{minipage}

\begin{minipage}[t]{0.5\textwidth}
\fbox{$e \rrFD A$} extends $e \rrD A$:
\begin{mathpar}
  \inferrule*{
    v_{f} = \vlam{x}{e}
  }{
    (\vmonfun{(\tarr{\tau_d}{\tau_c})}{v_f})~v_{arg} \rrFD \vsubst{e}{x}{v_{arg}}
  }

  \inferrule*{
    v_{f} = \vlam{\tann{x}{\tau}}{e}
    \\
    v_{arg}' = \fromany(\tau, v_{arg})
  }{
    (\vmonfun{(\tarr{\tau_d}{\tau_c})}{v_f})~v_{arg} \rrFD \esta{\tau_c}{v_f~v_{arg}'}
  }

  \inferrule*{
    v_0' = \fromany(\tau_0, v_0)
  }{
    \efst{(\vmonpair{(\tpair{\tau_0}{\tau_1})}{\vpair{v_0}{v_1}})} \rrFD v_0'
  }

  \inferrule*{
    v_1' = \fromany(\tau_1, v_1)
  }{
    \esnd{(\vmonpair{(\tpair{\tau_0}{\tau_1})}{\vpair{v_0}{v_1}})} \rrFD v_1'
  }
\end{mathpar}
\end{minipage}%
\begin{minipage}[t]{0.5\textwidth}
\fbox{$e \rrFS A$} extends $e \rrS A$:
\begin{mathpar}
  \inferrule*{
    v_{f} = \vlam{\tann{x}{\tau}}{e}
    \\
    v_{arg}' = \fromany(\tau, v_{arg})
  }{
    (\vmonfun{(\tarr{\tau_d}{\tau_c})}{v_f})~v_{arg} \rrFS v_f~v_{arg}'
  }

  \inferrule*{
    v_{f} = \vlam{x}{e}
    \\
    e_{arg}' = \esta{\tau_d}{v_{arg}}
  }{
    (\vmonfun{(\tarr{\tau_d}{\tau_c})}{v_f})~v_{arg} \rrFS \edyn{\tau_c}{v_f~e_{arg}'}
  }

  \inferrule*{
    v_0' = \fromany(\tau_0, v_0)
  }{
    \efst{(\vmonpair{(\tpair{\tau_0}{\tau_1})}{\vpair{v_0}{v_1}})} \rrFS v_0'
  }

  \inferrule*{
    v_1' = \fromany(\tau_1, v_1)
  }{
    \esnd{(\vmonpair{(\tpair{\tau_0}{\tau_1})}{\vpair{v_0}{v_1}})} \rrFS v_1'
  }
\end{mathpar}
\end{minipage}%
