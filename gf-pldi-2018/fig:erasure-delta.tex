$\begin{array}{l@{~~}c@{~~}l}
  \ED & \Coloneqq & \ldots \mid \esta{\tau}{\ED} \mid \edyn{\tau}{\ED}
\end{array}$

\fbox{$\Gamma \wellEE e$} inherits $\Gamma \wellD e$:
\begin{mathpar}
  \inferrule*{
    x,\Gamma \wellEE e
  }{
    \Gamma \wellEE \vlam{\tann{x}{\tau}}{e}
  }

  \inferrule*{
    \Gamma \wellEE e
  }{
    \Gamma \wellEE \edyn{\tau}{e}
  }

  \inferrule*{
    \Gamma \wellEE e
  }{
    \Gamma \wellEE \esta{\tau}{e}
  }
\end{mathpar}

$\begin{array}{l@{~~}c@{~~}l}

  \edyn{\tau}{v} & \rrEE & v
\\
  \esta{\tau}{v} & \rrEE & v
\end{array}$
