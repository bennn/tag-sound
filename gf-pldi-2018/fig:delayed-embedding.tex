\begin{flushleft}

\fbox{Language $\langL$} extends $\langN$:\\
$\begin{array}{l c l}
  v & = & \ldots \mid \vmonpair{(\tpair{\tau}{\tau})}{v}
\end{array}$

%% -----------------------------------------------------------------------------
\medskip
\begin{minipage}[t]{0.5\textwidth}
\fbox{$\Gamma \wellLE e$} extends $\Gamma \wellNE e$:
\begin{mathpar}
  \inferrule*{
    \Gamma \wellLE v : \tpair{\tau_0}{\tau_1}
  }{
    \Gamma \wellLE \vmonpair{(\tpair{\tau_0}{\tau_1})}{v}
  }
\end{mathpar}
\end{minipage}%
\begin{minipage}[t]{0.5\textwidth}
\fbox{$\Gamma \wellLE e : \tau$} extends $\Gamma \wellNE e : \tau$:
\begin{mathpar}
  \inferrule*{
    \Gamma \wellLE v
  }{
    \Gamma \wellLE \vmonfun{(\tpair{\tau_0}{\tau_1})}{v} : (\tpair{\tau_0}{\tau_1})
  }
\end{mathpar}
\end{minipage}

%% -----------------------------------------------------------------------------
\medskip
\begin{minipage}[t]{0.5\textwidth}
\fbox{$e \rrLD A$} extends $e \rrND A$:
\begin{mathpar}
  \inferrule*{
  }{
    \efst{(\vmonpair{(\tpair{\tau_0}{\tau_1})}{v})} \rrLD \esta{\tau_0}{\efst{v}}
  }

  \inferrule*{
  }{
    \esnd{(\vmonpair{(\tpair{\tau_0}{\tau_1})}{v})} \rrLD \esta{\tau_1}{\esnd{v}}
  }
\end{mathpar}
\end{minipage}%
\begin{minipage}[t]{0.5\textwidth}
\fbox{$e \rrLS A$} extends $e \rrNS A$:
\begin{mathpar}
  \inferrule*{
  }{
    \efst{(\vmonpair{(\tpair{\tau_0}{\tau_1})}{v})} \rrLS \edyn{\tau_0}{\efst{v}}
  }

  \inferrule*{
  }{
    \esnd{(\vmonpair{(\tpair{\tau_0}{\tau_1})}{v})} \rrLS \edyn{\tau_1}{\esnd{v}}
  }
\end{mathpar}
\end{minipage}

\fbox{$\wellSD{v}{\tau}{v}$}
\begin{mathpar}
  \inferrule*{
  }{
    \wellSD{v}{\tnat}{v}
  }

  \inferrule*{
  }{
    \wellSD{v}{\tint}{v}
  }

  \inferrule*{
  }{
    \wellSD{v}{(\tpair{\tau_0}{\tau_1})}{\vmonpair{(\tpair{\tau_0}{\tau_1})}{v}}
  }

  \inferrule*{
  }{
    \wellSD{v}{(\tarr{\tau_d}{\tau_c})}{\vmonfun{(\tarr{\tau_d}{\tau_c})}{v}}
  }
\end{mathpar}

\medskip
\fbox{$\wellDS{v}{\tau}{v}$}
\begin{mathpar}
  \inferrule*{
    v \in \naturals
  }{
    \wellDS{v}{\tnat}{v}
  }

  \inferrule*{
    v \in \integers
  }{
    \wellDS{v}{\tint}{v}
  }

  \inferrule*{
    v = \vpair{v_0}{v_1}
  }{
    \wellDS{v}{(\tpair{\tau_0}{\tau_1})}{\vmonpair{(\tpair{\tau_0}{\tau_1})}{v}}
  }

  \inferrule*{
    v = \vlam{x}{e}
    \\
    \vee
    \\
    v = \vmonfun{\tau}{v'}
  }{
    \wellDS{v}{(\tarr{\tau_d}{\tau_c})}{\vmonfun{(\tarr{\tau_d}{\tau_c})}{v}}
  }
\end{mathpar}
