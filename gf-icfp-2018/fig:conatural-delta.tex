  $\begin{array}{l@{\hspace{0.5em}}c@{\hspace{0.5em}}l@{\hspace{1em}}l}

    & \mbox{if $e \ccCS e'$}
    \\
    \ctxE{\esta{\tau}{e}} & \ccCD & A
    & \mbox{if $e \ccCS A$ and $A \not\in e$}

    \ctxE{e} & \ccCD & \ctxE{e'}
    & \mbox{if $e \rrCD e'$}
    \\
    \ctxE{e} & \ccCD & A
    & \mbox{if $e \rrCD A$ and $A \not\in e$}
  \end{array}$
