%% --- typing

\bigvspace

\lblmathpar{$\GammaD \wellM e$}{
  \inferrule*{
    x \in \GammaD
  }{
    \GammaD \wellM x
  }

  \inferrule*{
    x,\GammaD \wellM e
  }{
    \GammaD \wellM \vlam{x}{e}
  }

  \inferrule*{
  }{
    \GammaD \wellM i
  }

  \inferrule*{
    \GammaD \wellM e_0
    \\\\
    \GammaD \wellM e_1
  }{
    \GammaD \wellM \vpair{e_0}{e_1}
  }

  \inferrule*{
    \GammaD \wellM e_0
    \\\\
    \GammaD \wellM e_1
  }{
    \GammaD \wellM e_0~e_1
  }

  \inferrule*{
    \GammaD \wellM e
  }{
    \GammaD \wellM \vunop~e
  }

  \inferrule*{
    \GammaD \wellM e_0
    \\\\
    \GammaD \wellM e_1
  }{
    \GammaD \wellM \vbinop~e_0~e_1
  }

  \inferrule*{
    \Gamma \wellM e : \tau
  }{
    \Gamma \wellM \esta{\tau}{e}
  }
}

\smallskip
\lblmathpar{$\GammaS \wellM e : \tau$}{
  \inferrule*{
    \tann{x}{\tau} \in \GammaS
  }{
    \GammaS \wellM x : \tau
  }

  \inferrule*{
    \tann{x}{\tau_d},\GammaS \wellM e : \tau_c
  }{
    \GammaS \wellM \vlam{\tann{x}{\tau_d}}{e} : \tau_d \tarrow \tau_c
  }

  \inferrule*{
    i \in \naturals
  }{
    \GammaS \wellM i : \tnat
  }

  \inferrule*{
  }{
    \GammaS \wellM i : \tint
  }

  \inferrule*{
    \GammaS \wellM e_0 : \tau_0
    \\\\
    \GammaS \wellM e_1 : \tau_1
  }{
    \GammaS \wellM \vpair{e_0}{e_1} : \tpair{\tau_0}{\tau_1}
  }

  \inferrule*{
    \GammaS \wellM e_0 : \tau_d \tarrow \tau_c
    \\\\
    \GammaS \wellM e_1 : \tau_d
  }{
    \GammaS \wellM e_0~e_1 : \tau_c
  }

  \inferrule*{
    \GammaS \wellM e_0 : \tau_0
    \\\\
    \Delta(\vunop, \tau_0) = \tau
  }{
    \GammaS \wellM \vunop~e_0 : \tau
  }

  \inferrule*{
    \GammaS \wellM e_0 : \tau_0
    \\\\
    \GammaS \wellM e_1 : \tau_1
    \\\\
    \Delta(\vbinop, \tau_0, \tau_1) = \tau
  }{
    \GammaS \wellM \vbinop~e_0~e_1 : \tau
  }

  \inferrule*{
    \GammaS \wellM e : \tau'
    \\\\
    \tau' \subt \tau
  }{
    \GammaS \wellM e : \tau
  }

  \inferrule*{
    \Gamma \wellM e
  }{
    \Gamma \wellM \edyn{\tau}{e} : \tau
  }
}

\smallskip
\lblmathpar{$\tau \subt \tau$}{
  \inferrule*{
  }{
    \tnat \subt \tint
  }

  \inferrule*{
    \tau_d' \subt \tau_d
  }{
    \tarr{\tau_d}{\tau_c} \subt \tarr{\tau_d'}{\tau_c}
  }

  \inferrule*{
    \tau_c \subt \tau_c'
  }{
    \tarr{\tau_d}{\tau_c} \subt \tarr{\tau_d}{\tau_c'}
  }

  \inferrule*{
    \tau_0 \subt \tau_0'
  }{
    \tpair{\tau_0}{\tau_1} \subt \tpair{\tau_0'}{\tau_1}
  }

  \inferrule*{
    \tau_1 \subt \tau_1'
  }{
    \tpair{\tau_0}{\tau_1} \subt \tpair{\tau_0}{\tau_1'}
  }
}

\begin{multicols}{2}
\lbl{$\Delta(\vunop, \tau) = \tau$}{
$\begin{array}{l@{~~}c@{~~}l}
  \Delta(\vfst, \tpair{\tau_0}{\tau_1}) & = & \tau_0
\\
  \Delta(\vsnd, \tpair{\tau_0}{\tau_1}) & = & \tau_1
\end{array}$
}

\lbl{$\Delta(\vbinop, \tau, \tau) = \tau$}{
$\begin{array}{l@{~~}c@{~~}l}
  \Delta(\vbinop, \tnat, \tnat) & = & \tnat
\\
  \Delta(\vbinop, \tint, \tint) & = & \tint
\end{array}$
}

\vfill\null\columnbreak

\lbl{$\delta(\vunop, v) = A$}{
$\begin{array}{l@{~~}c@{~~}l}
  \delta(\vfst, \vpair{v_0}{v_1}) & = & v_0
\\
  \delta(\vsnd, \vpair{v_0}{v_1}) & = & v_1
\end{array}$
}

\lbl{$\delta(\vbinop, v, v) = A$}{
$\begin{array}{l@{~~}c@{~~}l}
  \delta(\vsum, i_0, i_1) & = & i_0 + i_1
\\
  \delta(\vquotient, i_0, 0) & = & \boundaryerror
\\
  \delta(\vquotient, i_0, i_1) & = & \floorof{i_0 / i_1}
\\ \multicolumn{3}{l}{\mbox{\quad if $i_1 \neq 0$}}
\end{array}$
}
\end{multicols}
