\lbl{Common Syntax}{
$\begin{array}{l@{~~}c@{~~}l}
  e & \BNFeq & x \mid v \mid e~e \mid \vpair{e}{e} \mid \vunop~e \mid \vbinop~e~e
\\
  v & \BNFeq & i \mid \vpair{v}{v}
\\
  \tau & \BNFeq & \tint \mid \tnat \mid \tpair{\tau}{\tau} \mid \tau \tarrow \tau
\\
  i & \BNFeq & 0 \mid 1 \mid -1 \mid \ldots
\\
  \vunop & \BNFeq & \vfst \mid \vsnd
\\
  \vbinop & \BNFeq & \vsum \mid \vquotient
\\
  A & \BNFeq & e \mid \boundaryerror \mid \tagerror
\\
  E & \BNFeq & \ehole \mid \ED\,e \mid v\,\ED \mid \vpair{\ED}{\!e} \mid \vpair{v}{\!\ED} \mid
 \vunop\,\ED \mid \vbinop\,\!\ED\,e \mid \vbinop\,\!v\,\ED
\end{array}$
}

\smallskip
\lbl{$\delta(\vunop, v) = A$}{
$\begin{array}{l@{~~}c@{~~}l}
  \delta(\vfst, \vpair{v_0}{v_1}) & = & v_0
\\
  \delta(\vsnd, \vpair{v_0}{v_1}) & = & v_1
\end{array}$
}

\smallskip
\lbl{$\delta(\vbinop, v, v) = A$}{
$\begin{array}{l@{~~}c@{~~}l l}
  \delta(\vsum, i_0, i_1) & = & i_0 + i_1 &
\\
  \delta(\vquotient, i_0, 0) & = & \boundaryerror &
\\
  \delta(\vquotient, i_0, i_1) & = & \floorof{i_0 / i_1} & \mbox{\quad if $i_1 \neq 0$}
\end{array}$
}

\smallskip
\lbl{$\Delta(\vunop, \tau) = \tau$}{
$\begin{array}{l@{~~}c@{~~}l}
  \Delta(\vfst, \tpair{\tau_0}{\tau_1}) & = & \tau_0
\\
  \Delta(\vsnd, \tpair{\tau_0}{\tau_1}) & = & \tau_1
\end{array}$
}

\smallskip
\lbl{$\Delta(\vbinop, \tau, \tau) = \tau$}{
$\begin{array}{l@{~~}c@{~~}l}
  \Delta(\vbinop, \tnat, \tnat) & = & \tnat
\\
  \Delta(\vbinop, \tint, \tint) & = & \tint
\end{array}$
}

\smallskip
\lblmathpar{$\tau \subt \tau$}{
  \inferrule*{
  }{
    \tnat \subt \tint
  }

  \inferrule*{
    \tau_d' \subt \tau_d
  }{
    \tarr{\tau_d}{\tau_c} \subt \tarr{\tau_d'}{\tau_c}
  }

  \inferrule*{
    \tau_c \subt \tau_c'
  }{
    \tarr{\tau_d}{\tau_c} \subt \tarr{\tau_d}{\tau_c'}
  }

  \inferrule*{
    \tau_0 \subt \tau_0'
  }{
    \tpair{\tau_0}{\tau_1} \subt \tpair{\tau_0'}{\tau_1}
  }

  \inferrule*{
    \tau_1 \subt \tau_1'
  }{
    \tpair{\tau_0}{\tau_1} \subt \tpair{\tau_0}{\tau_1'}
  }
}

\lbl{$e \cclift{E}{\rrR} A$}{
$\begin{array}{l@{~~}c@{~~}l l}
\ctxE{e} & \cclift{E}{\rrR} & \ctxE{e'} & \mbox{if $e \rrR e'$}
\\
\ctxE{e} & \cclift{E}{\rrR} & A         & \mbox{if $e \rrR A$ and $A \not\in e$}
\end{array}$
}

\end{flushleft}
