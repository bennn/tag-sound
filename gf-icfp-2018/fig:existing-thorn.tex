\lblextra{$\vfromsta : \specarrp{\tpair{\tau}{v}}{e}$}{}{
  $\begin{array}{l@{~~}c@{~}l}
    \efromsta{C}{v} & = & p
    \\ \sidecond{if $v \valeq p$ or $v \valeq \thorncast{\thorndyn}{p}$ or $v \valeq \thorncast{\thornlike{C'}}{p}$}
    \\ \sidecond{and $p \valeq C''(f = v, \ldots)$}
    \\ \sidecond{and $C'' \subteq C$}
    \\
    \efromsta{\thornlike{C}}{v} & = & \thorncast{\thornlike{C}}{p}
    \\ \sidecond{if $v \valeq p$ or $v \valeq \thorncast{\thorndyn}{p}$ or $v \valeq \thorncast{\thornlike{C'}}{p}$}
    \\
    \efromsta{\thorndyn}{v} & = & \thorncast{\thorndyn}{p}
    \\ \sidecond{if $v \valeq p$ or $v \valeq \thorncast{\thorndyn}{p}$ or $v \valeq \thorncast{\thornlike{C'}}{p}$}
    \\
    \efromsta{\tau}{v} & = & \boundaryerror
    \\ \sidecond{otherwise}
  \end{array}$
}
