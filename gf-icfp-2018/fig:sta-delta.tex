\lblextra{Language $\langS$}{ extends Common Syntax:}{
$\begin{array}{l@{~~}c@{~~}l}
  v & \BNFeq & \ldots \mid \vlam{\tann{x}{\tau}}{e}
\\
  \GammaS & \BNFeq & \cdot \mid \tann{x}{\tau},\GammaS
\end{array}$
}

\smallskip
\lblmathpar{$\GammaS \wellsta e : \tau$}{
  \inferrule*{
    \tann{x}{\tau} \in \GammaS
  }{
    \GammaS \wellsta x : \tau
  }

  \inferrule*{
    \tann{x}{\tau_d},\GammaS \wellsta e : \tau_c
  }{
    \GammaS \wellsta \vlam{\tann{x}{\tau_d}}{e} : \tau_d \tarrow \tau_c
  }

  \inferrule*{
    i \in \naturals
  }{
    \GammaS \wellsta i : \tnat
  }

  \inferrule*{
  }{
    \GammaS \wellsta i : \tint
  }

  \inferrule*{
    \GammaS \wellsta e_0 : \tau_0
    \\\\
    \GammaS \wellsta e_1 : \tau_1
  }{
    \GammaS \wellsta \vpair{e_0}{e_1} : \tpair{\tau_0}{\tau_1}
  }

  \inferrule*{
    \GammaS \wellsta e_0 : \tau_d \tarrow \tau_c
    \\\\
    \GammaS \wellsta e_1 : \tau_d
  }{
    \GammaS \wellsta e_0~e_1 : \tau_c
  }

  \inferrule*{
    \GammaS \wellsta e_0 : \tau_0
    \\\\
    \Delta(\vunop, \tau_0) = \tau
  }{
    \GammaS \wellsta \vunop~e_0 : \tau
  }

  \inferrule*{
    \GammaS \wellsta e_0 : \tau_0
    \\\\
    \GammaS \wellsta e_1 : \tau_1
    \\\\
    \Delta(\vbinop, \tau_0, \tau_1) = \tau
  }{
    \GammaS \wellsta \vbinop~e_0~e_1 : \tau
  }

  \inferrule*{
    \GammaS \wellsta e : \tau'
    \\\\
    \tau' \subt \tau
  }{
    \GammaS \wellsta e : \tau
  }
}

\smallskip
\lbl{$e \rrS A$}{
$\begin{array}{l@{~~}c@{~~}l}
  (\vlam{\tann{x}{\tau_d}}{e})~v & \rrS & \vsubst{e}{x}{v}
\\
  \eunop{v} & \rrS & \delta(\vunop, v)
\\
  \ebinop{v_0}{v_1} & \rrS & \delta(\vbinop, v_0, v_1)
\end{array}$
}

\smallskip
\fbox{$e \rrSstar A$} reflexive, transitive closure of $\cclift{\ES}{\rrS}$
